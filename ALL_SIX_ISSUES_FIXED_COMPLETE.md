```sql

SELECT t.text, tt.tag, tt.is_auto
FROM todos t
JOIN todo_tags tt ON t.id = tt.todo_id
ORDER BY t.text, tt.is_auto DESC, tt.tag;

-- Verify inheritance
SELECT 
    t.text,
    t.category_id,
    t.source_note_id,
    GROUP_CONCAT(tt.tag, ', ') as tags
FROM todos t
LEFT JOIN todo_tags tt ON t.id = tt.todo_id
WHERE t.source_type = 'note'
GROUP BY t.id;
```
